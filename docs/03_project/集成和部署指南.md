### 通过 Python 代码调用

```python
from ai.tools import dispatch_tool
from models import async_session

async def verify_knowledge(topic: str):
    """验证知识库中的内容."""
    async with async_session() as session:
        result = await dispatch_tool(
            session,

            {"topic": topic}
```

### 批量操作

```python
# 同时调用多个工具（异步执行）
import asyncio

async def verify_response(statement: str):
    """完整的验证流程."""
    
    tasks = [
        dispatch_tool(session, "knowledge_verify", {"topic": statement}),
        dispatch_tool(session, "fact_checker", {"claim": statement}),
        dispatch_tool(session, "confidence_score", 
                     {"statement": statement, "source": "learned"}),
    ]
    
    results = await asyncio.gather(*tasks)
    return results
```
