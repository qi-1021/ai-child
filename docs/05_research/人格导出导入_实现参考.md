```python
"""
Profile management endpoints for export, import, and backup.
"""
import json
import logging
from datetime import datetime
from io import BytesIO
from typing import Optional

from fastapi import APIRouter, Depends, File, HTTPException, UploadFile
from sqlalchemy.ext.asyncio import AsyncSession

from ai.memory import get_all_knowledge, get_recent_messages
from ai.profile import get_ai_name, get_or_create_profile
from ai.tools import get_all_tools
from models import get_session, KnowledgeItem, Conversation, Tool
from models.schemas import MessageOut, KnowledgeOut, ToolOut

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/profile", tags=["profile"])

# ─── Export Format Version ────────────────────────────────────────────

EXPORT_FORMAT_VERSION = "1.0"
COMPATIBLE_VERSIONS = ["1.0"]  # 支持的导入版本


# ─── Data Schema Classes ──────────────────────────────────────────────

class ProfileSnapshot:
    """AI 人格快照"""
    
    def __init__(self, profile):
        self.name = profile.name
        self.created_at = profile.created_at
        self.named_at = profile.named_at
        self.is_sleeping = profile.is_sleeping
        self.last_consolidation_summary = profile.last_consolidation_summary
    
    def to_dict(self):
        return {
            "name": self.name,
            "created_at": self.created_at.isoformat() if self.created_at else None,
            "named_at": self.named_at.isoformat() if self.named_at else None,
            "is_sleeping": self.is_sleeping,
            "last_consolidation_summary": self.last_consolidation_summary,
        }


class ExportMetadata:
    """导出元数据"""
    
    def __init__(self, backup_type: str = "full"):
        self.export_version = EXPORT_FORMAT_VERSION
        self.app_version = "Phase1"
        self.export_at = datetime.utcnow()
        self.backup_type = backup_type  # "full" | "profile_only" | "metadata"
    
    def to_dict(self):
        return {
            "export_version": self.export_version,
            "app_version": self.app_version,
            "export_at": self.export_at.isoformat(),
            "backup_type": self.backup_type,
        }


# ─── Export Endpoint ──────────────────────────────────────────────────

@router.get("/export")
async def export_profile(
    include_data: str = "full",  # "full" | "profile_only" | "metadata"
    session: AsyncSession = Depends(get_session)
):
    """
    Export AI personality snapshot as JSON.
    
    Args:
        include_data: 导出内容类型
          - 'full': 完整备份（人格+数据+元数据）
          - 'profile_only': 仅人格信息
          - 'metadata': 仅统计信息
    
    Returns:
        导出的 JSON 结构
    
    Examples:
        GET /profile/export?include_data=full
        GET /profile/export?include_data=profile_only
    """
    try:
        profile = await get_or_create_profile(session)
        
        # 1. 收集人格信息
        profile_data = ProfileSnapshot(profile).to_dict()
        
        # 2. 收集统计信息
        conversations = await get_recent_messages(session, limit=None)
        knowledge = await get_all_knowledge(session)
        tools = await get_all_tools(session)
        
        statistics = {
            "total_conversations": len(conversations),
            "total_knowledge_items": len(knowledge),
            "total_tools": len(tools),
            "ai_named": profile.name is not None,
            "is_sleeping": profile.is_sleeping,
        }
        
        # 3. 构造导出对象
        export_data = {
            "format": "ai-child-backup",
            "profile": profile_data,
            "statistics": statistics,
            "metadata": ExportMetadata(backup_type=include_data).to_dict(),
        }
        
        # 4. 根据类型添加详细数据
        if include_data == "full":
            export_data["data"] = {
                "conversations": [
                    {
                        "id": m.id,
                        "role": m.role,
                        "content": m.content,
                        "content_type": m.content_type,
                        "timestamp": m.timestamp.isoformat() if m.timestamp else None,
                        "media_path": m.media_path,
                    }
                    for m in conversations
                ],
                "knowledge": [
                    {
                        "id": k.id,
                        "topic": k.topic,
                        "content": k.content,
                        "source": k.source,
                        "confidence": k.confidence,
                        "timestamp": k.timestamp.isoformat() if k.timestamp else None,
                    }
                    for k in knowledge
                ],
                "tools": [
                    {
                        "id": t.id,
                        "name": t.name,
                        "description": t.description,
                        "code": t.code,
                        "parameters_schema": t.parameters_schema,
                        "call_count": t.call_count,
                        "created_at": t.created_at.isoformat() if t.created_at else None,
                    }
                    for t in tools
                ],
            }
        
        logger.info(f"Profile exported: {profile.name or 'unnamed'}, type={include_data}")
        return export_data
    
    except Exception as e:
        logger.error(f"Export failed: {e}")
        raise HTTPException(status_code=500, detail=f"Export failed: {str(e)}")


# ─── Import Endpoint ──────────────────────────────────────────────────

@router.post("/import")
async def import_profile(
    file: UploadFile = File(...),
    session: AsyncSession = Depends(get_session)
):
    """
    Import AI personality from exported JSON backup.
    
    Args:
        file: 导出的 JSON 备份文件
    
    Returns:
        导入结果
    
    Examples:
        POST /profile/import
        Content-Type: multipart/form-data
        file: <exported_backup.json>
    """
    try:
        # 1. 读取文件
        content = await file.read()
        import_data = json.loads(content.decode('utf-8'))
        
        # 2. 验证文件格式
        if import_data.get("format") != "ai-child-backup":
            raise ValueError("Invalid backup format")
        
        # 3. 验证版本兼容性
        version = import_data.get("metadata", {}).get("export_version")
        if version not in COMPATIBLE_VERSIONS:
            raise ValueError(f"Incompatible version: {version}. Supported: {COMPATIBLE_VERSIONS}")
        
        # 4. 验证必需字段
        if "profile" not in import_data:
            raise ValueError("Missing 'profile' field in backup")
        
        # 5. 恢复人格信息
        profile = await get_or_create_profile(session)
        profile_data = import_data["profile"]
        
        if profile_data.get("name"):
            profile.name = profile_data["name"]
            logger.info(f"Profile name restored: {profile.name}")
        
        if profile_data.get("named_at"):
            from dateutil.parser import parse
            profile.named_at = parse(profile_data["named_at"])
        
        profile.is_sleeping = profile_data.get("is_sleeping", False)
        
        if profile_data.get("last_consolidation_summary"):
            profile.last_consolidation_summary = profile_data["last_consolidation_summary"]
        
        await session.commit()
        
        # 6. 返回导入摘要
        return {
            "status": "success",
            "profile_restored": {
                "name": profile.name,
                "named_at": profile.named_at,
            },
            "statistics": import_data.get("statistics", {}),
            "message": f"Profile '{profile.name}' restored successfully",
        }
    
    except json.JSONDecodeError:
        logger.error("Invalid JSON in backup file")
        raise HTTPException(status_code=400, detail="Invalid JSON format")
    
    except ValueError as e:
        logger.error(f"Validation error: {e}")
        raise HTTPException(status_code=400, detail=f"Validation error: {str(e)}")
    
    except Exception as e:
        logger.error(f"Import failed: {e}")
        raise HTTPException(status_code=500, detail=f"Import failed: {str(e)}")


# ─── Backup Management ────────────────────────────────────────────────

@router.post("/backup")
async def create_backup(session: AsyncSession = Depends(get_session)):
    """
    Create a named backup of current profile state.
    
    Returns:
        备份信息
    """
    try:
        profile = await get_or_create_profile(session)
        backup_name = f"backup_{profile.name or 'unnamed'}_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.json"
        
        # 通过 export 生成备份
        export_response = await export_profile(include_data="full", session=session)
        
        logger.info(f"Backup created: {backup_name}")
        return {
            "status": "success",
            "backup_name": backup_name,
            "size_bytes": len(json.dumps(export_response)),
            "created_at": datetime.utcnow().isoformat(),
        }
    
    except Exception as e:
        logger.error(f"Backup creation failed: {e}")
        raise HTTPException(status_code=500, detail=f"Backup failed: {str(e)}")


# ─── Info/Status Endpoint ─────────────────────────────────────────────

@router.get("/info")
async def profile_info(session: AsyncSession = Depends(get_session)):
    """
    Get current profile status and info.
    
    Returns:
        人格状态信息
    """
    try:
        profile = await get_or_create_profile(session)
        conversations = await get_recent_messages(session, limit=None)
        knowledge = await get_all_knowledge(session)
        
        return {
            "name": profile.name,
            "has_name": profile.name is not None,
            "created_at": profile.created_at.isoformat() if profile.created_at else None,
            "named_at": profile.named_at.isoformat() if profile.named_at else None,
            "is_sleeping": profile.is_sleeping,
            "last_consolidation_summary": profile.last_consolidation_summary,
            "stats": {
                "conversations": len(conversations),
                "knowledge_items": len(knowledge),
            }
        }
    
    except Exception as e:
        logger.error(f"Profile info retrieval failed: {e}")
        raise HTTPException(status_code=500, detail=f"Failed to get profile info: {str(e)}")
```

```python
"""Test profile export/import functionality."""
import json
import httpx
import asyncio
from pathlib import Path

BASE_URL = "http://localhost:8000"


async def test_export_import():
    """测试导出和导入流程。"""
    
    async with httpx.AsyncClient() as client:
        # 1. 获取当前状态
        print("1️⃣  获取人格信息...")
        resp = await client.get(f"{BASE_URL}/profile/info")
        print(f"   当前人格: {resp.json()}")
        
        # 2. 导出备份
        print("\n2️⃣  导出完整备份...")
        resp = await client.get(
            f"{BASE_URL}/profile/export",
            params={"include_data": "full"}
        )
        export_data = resp.json()
        print(f"   导出成功，包含 {export_data['statistics']['total_conversations']} 条对话")
        
        # 3. 保存到文件
        backup_path = Path("backup.json")
        with open(backup_path, "w") as f:
            json.dump(export_data, f, indent=2, default=str)
        print(f"   已保存到 {backup_path}")
        
        # 4. 导入备份（模拟恢复场景）
        print("\n3️⃣  测试导入备份...")
        with open(backup_path, "rb") as f:
            files = {"file": ("backup.json", f, "application/json")}
            resp = await client.post(f"{BASE_URL}/profile/import", files=files)
            import_result = resp.json()
            print(f"   导入结果: {import_result['message']}")
        
        # 5. 验证导入后的状态
        print("\n4️⃣  验证导入后状态...")
        resp = await client.get(f"{BASE_URL}/profile/info")
        restored = resp.json()
        print(f"   恢复的人格: {restored['name']}")
        
        return True


if __name__ == "__main__":
    asyncio.run(test_export_import())
```

### 1. 添加数据加密

```python
from cryptography.fernet import Fernet
import os

def encrypt_backup(data: dict, password: str) -> str:
    """加密备份数据。"""
    key = Fernet.generate_key()
    cipher = Fernet(key)
    json_str = json.dumps(data, default=str)
    encrypted = cipher.encrypt(json_str.encode())
    return encrypted.decode()


def decrypt_backup(encrypted: str, key: str) -> dict:
    """解密备份数据。"""
    cipher = Fernet(key)
    decrypted = cipher.decrypt(encrypted.encode())
    return json.loads(decrypted.decode())
```
